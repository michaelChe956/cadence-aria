### Phase 1：消息分组工具函数与测试

**目标**：实现 `groupEntries()` 分组逻辑，纯函数，先写测试。

#### Step 1.1：创建分组类型定义

文件：`web/src/components/chat-workspace/message-grouping.ts`

```typescript
import type { ChatEntry, ChatEntryRole } from "../../state/chat-entries";

export interface MessageGroup {
  id: string;
  node_id: string;
  role: ChatEntryRole;
  primaryEntry: ChatEntry | null;
  inlineEvents: ChatEntry[];
  interruptEntries: ChatEntry[];
}

const SYSTEM_ENTRY_TYPES = new Set([
  "stage_change",
  "artifact_update",
  "review_verdict",
  "start_generation",
  "context_note",
  "error",
  "human_decision",
]);

export type GroupedItem =
  | { kind: "group"; group: MessageGroup }
  | { kind: "standalone"; entry: ChatEntry };

export function groupEntries(entries: ChatEntry[]): GroupedItem[] {
  const result: GroupedItem[] = [];
  let currentGroup: MessageGroup | null = null;

  for (const entry of entries) {
    if (SYSTEM_ENTRY_TYPES.has(entry.type)) {
      if (currentGroup) {
        result.push({ kind: "group", group: currentGroup });
        currentGroup = null;
      }
      result.push({ kind: "standalone", entry });
      continue;
    }

    const nodeId = entry.node_id ?? null;
    if (!nodeId) {
      if (currentGroup) {
        result.push({ kind: "group", group: currentGroup });
        currentGroup = null;
      }
      result.push({ kind: "standalone", entry });
      continue;
    }

    if (!currentGroup || currentGroup.node_id !== nodeId) {
      if (currentGroup) {
        result.push({ kind: "group", group: currentGroup });
      }
      currentGroup = {
        id: `group:${nodeId}`,
        node_id: nodeId,
        role: entry.role,
        primaryEntry: null,
        inlineEvents: [],
        interruptEntries: [],
      };
    }

    if (entry.type === "provider_stream") {
      currentGroup.primaryEntry = entry;
      currentGroup.role = entry.role;
    } else if (entry.type === "execution_event") {
      currentGroup.inlineEvents.push(entry);
    } else if (entry.type === "permission_request" || entry.type === "gate_prompt") {
      currentGroup.interruptEntries.push(entry);
    } else {
      currentGroup.inlineEvents.push(entry);
    }
  }

  if (currentGroup) {
    result.push({ kind: "group", group: currentGroup });
  }

  return result;
}
```

```typescript
import { useState } from "react";
import { ChevronRight, Wrench } from "lucide-react";
import type { ChatEntry } from "../../state/chat-entries";

interface InlineEventRowProps {
  entry: ChatEntry;
}

export function InlineEventRow({ entry }: InlineEventRowProps) {
  const [expanded, setExpanded] = useState(false);
  const event = entry.metadata as Record<string, unknown> | undefined;
  const title = entry.content || "执行事件";

  return (
    <div className="my-1.5 rounded border border-[var(--aria-line)] bg-[var(--aria-panel-muted)]">
      <button
        type="button"
        onClick={() => setExpanded(!expanded)}
        className="flex w-full items-center gap-1.5 px-2 py-1.5 text-left text-xs text-[var(--aria-ink-muted)] hover:bg-[var(--aria-panel)]"
      >
        <ChevronRight
          className={`h-3 w-3 shrink-0 transition-transform ${expanded ? "rotate-90" : ""}`}
        />
        <Wrench className="h-3 w-3 shrink-0 text-[var(--aria-primary)]" />
        <span className="truncate font-medium">{title}</span>
      </button>
      {expanded && (
        <div className="border-t border-[var(--aria-line)] px-2 py-1.5">
          {event?.command && typeof event.command === "string" ? (
            <div className="rounded bg-white px-2 py-1 font-mono text-xs text-[var(--aria-ink-muted)]">
              {event.command}
            </div>
          ) : null}
          {event?.output && typeof event.output === "string" ? (
            <pre className="mt-1 max-h-40 overflow-auto whitespace-pre-wrap rounded bg-white px-2 py-1 font-mono text-xs text-[var(--aria-ink)]">
              {event.output}
            </pre>
          ) : null}
        </div>
      )}
    </div>
  );
}
```

```typescript
import type { MessageGroup } from "./message-grouping";
import type { ChatEntry } from "../../state/chat-entries";
import type { RevisionPath } from "../../api/types";
import { ChatEntryContainer } from "./ChatEntryContainer";
import { InlineEventRow } from "./InlineEventRow";
import { ChatEntryRenderer } from "./ChatEntryRenderer";

interface MessageGroupViewProps {
  group: MessageGroup;
  onPermissionResponse?: (entry: ChatEntry, approved: boolean) => void;
  onSelectRevisionPath?: (path: RevisionPath) => void;
  onHumanConfirm?: (decision: "confirm" | "request-change" | "terminate") => void;
}

export function MessageGroupView({
  group,
  onPermissionResponse,
  onSelectRevisionPath,
  onHumanConfirm,
}: MessageGroupViewProps) {
  const roleLabel = group.role === "reviewer" ? "审核者" : "作者";
  const containerRole = group.role === "reviewer" ? "reviewer" : "author";

  return (
    <ChatEntryContainer role={containerRole} title={roleLabel}>
      {group.primaryEntry && <StreamContent content={group.primaryEntry.content} />}

      {group.inlineEvents.length > 0 && (
        <div className="my-2">
          {group.inlineEvents.map((event) => (
            <InlineEventRow key={event.id} entry={event} />
          ))}
        </div>
      )}

      {group.interruptEntries.map((entry) => (
        <div key={entry.id} className="my-2">
          <ChatEntryRenderer
            entry={entry}
            onPermissionResponse={onPermissionResponse}
            onSelectRevisionPath={onSelectRevisionPath}
            onHumanConfirm={onHumanConfirm}
          />
        </div>
      ))}
    </ChatEntryContainer>
  );
}

function StreamContent({ content }: { content: string }) {
  const blocks = content.split(/\n{2,}/).filter((block) => block.trim().length > 0);
  if (blocks.length === 0) {
    return <div className="whitespace-pre-wrap text-sm text-[var(--aria-ink)]" />;
  }

  return (
    <div className="space-y-2">
      {blocks.map((block, index) => {
        const trimmed = block.trim();
        if (trimmed.startsWith("### ")) {
          return (
            <h3 key={index} className="text-sm font-semibold text-[var(--aria-ink)]">
              {trimmed.slice(4)}
            </h3>
          );
        }
        if (trimmed.startsWith("## ")) {
          return (
            <h2 key={index} className="text-base font-semibold text-[var(--aria-ink)]">
              {trimmed.slice(3)}
            </h2>
          );
        }
        if (trimmed.startsWith("# ")) {
          return (
            <h1 key={index} className="text-lg font-semibold text-[var(--aria-ink)]">
              {trimmed.slice(2)}
            </h1>
          );
        }
        return (
          <div key={index} className="whitespace-pre-wrap text-sm text-[var(--aria-ink)]">
            {block}
          </div>
        );
      })}
    </div>
  );
}
```

**目标**：将 ChatEntryList 从平铺渲染改为分组渲染。

#### Step 4.1：修改 ChatEntryList.tsx

2. 在渲染前调用 `useMemo(() => groupEntries(entries), [entries])` 生成分组
3. 遍历 `GroupedItem[]`：
   - `kind === "group"` → 渲染 `<MessageGroupView>`

4. `data-entry-id` 使用 group.id 或 entry.id

```typescript
// ChatEntryList.tsx 核心改动
import { useMemo } from "react";
import { groupEntries } from "./message-grouping";
import { MessageGroupView } from "./MessageGroupView";

// 在组件内部：
const grouped = useMemo(() => groupEntries(entries), [entries]);

// 渲染：
{grouped.map((item) => {
  if (item.kind === "group") {
    return (
      <div key={item.group.id} data-entry-id={item.group.id} className="min-w-0">
        <MessageGroupView
          group={item.group}
          onPermissionResponse={onPermissionResponse}
          onSelectRevisionPath={onSelectRevisionPath}
          onHumanConfirm={onHumanConfirm}
        />
      </div>
    );
  }
  return (
    <div key={item.entry.id} data-entry-id={item.entry.id} className="min-w-0">
      <ChatEntryRenderer
        entry={item.entry}
        onPermissionResponse={onPermissionResponse}
        onSelectRevisionPath={onSelectRevisionPath}
        onHumanConfirm={onHumanConfirm}
      />
    </div>
  );
})}
```

```rust
pub fn extract_artifact_content(full_output: &str) -> String {
    if let Some(start) = full_output.find("<artifact>") {
        let content_start = start + "<artifact>".len();
        if let Some(end) = full_output[content_start..].find("</artifact>") {
            return full_output[content_start..content_start + end].trim().to_string();
        }
        return full_output[content_start..].trim().to_string();
    }

    if let Some(heading_pos) = full_output.find("\n# ") {
        return full_output[heading_pos..].trim().to_string();
    }
    if full_output.starts_with("# ") {
        return full_output.to_string();
    }

    full_output.to_string()
}
```
