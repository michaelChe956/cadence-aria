### Phase 4：前端 ChoiceRequest 组件

**目标**：渲染选择卡片 UI。

#### Step 4.1：创建 ChoiceRequestEntry 组件

文件：`web/src/components/chat-workspace/entries/ChoiceRequestEntry.tsx`

```typescript
import { useState } from "react";
import { ListChecks } from "lucide-react";
import type { ChatEntry } from "../../../state/chat-entries";
import { ChatEntryContainer } from "../ChatEntryContainer";

interface ChoiceRequestEntryProps {
  entry: ChatEntry;
  onChoiceResponse?: (id: string, selectedIndices: number[], textInput?: string) => void;
}

export function ChoiceRequestEntry({ entry, onChoiceResponse }: ChoiceRequestEntryProps) {
  const metadata = entry.metadata as Record<string, unknown> | undefined;
  const requestId = metadata?.request_id as string | undefined;
  const options = (metadata?.options ?? []) as Array<{ label: string; description?: string }>;
  const multiSelect = (metadata?.multi_select ?? false) as boolean;
  const resolved = entry.resolved ?? false;

  const [selected, setSelected] = useState<Set<number>>(new Set());
  const [textInput, setTextInput] = useState("");

  const toggleOption = (index: number) => {
    setSelected((prev) => {
      const next = new Set(prev);
      if (multiSelect) {
        if (next.has(index)) next.delete(index);
        else next.add(index);
      } else {
        next.clear();
        next.add(index);
      }
      return next;
    });
  };

  const handleSubmit = () => {
    if (!requestId || !onChoiceResponse) return;
    const indices = Array.from(selected).sort();
    onChoiceResponse(requestId, indices, textInput.trim() || undefined);
  };

  return (
    <ChatEntryContainer role="system" title="选择请求" testId="choice-request-entry">
      <div className="space-y-3">
        <div className="flex items-start gap-2 text-sm text-[var(--aria-ink)]">
          <ListChecks className="mt-0.5 h-4 w-4 shrink-0 text-[var(--aria-primary)]" />
          <div className="font-medium">{entry.content}</div>
        </div>

        {options.length > 0 && (
          <div className="space-y-1.5 pl-6">
            {options.map((option, index) => (
              <label
                key={index}
                className={`flex cursor-pointer items-start gap-2 rounded-md border px-3 py-2 text-sm transition-colors ${
                  selected.has(index)
                    ? "border-[var(--aria-primary)] bg-[var(--aria-primary)]/5"
                    : "border-[var(--aria-line)] hover:bg-[var(--aria-panel)]"
                } ${resolved ? "pointer-events-none opacity-60" : ""}`}
              >
                <input
                  type={multiSelect ? "checkbox" : "radio"}
                  name={`choice-${requestId}`}
                  checked={selected.has(index)}
                  onChange={() => toggleOption(index)}
                  disabled={resolved}
                  className="mt-0.5"
                />
                <div>
                  <div className="font-medium text-[var(--aria-ink)]">{option.label}</div>
                  {option.description && (
                    <div className="text-xs text-[var(--aria-ink-muted)]">{option.description}</div>
                  )}
                </div>
              </label>
            ))}
          </div>
        )}

        <div className="pl-6">
          <input
            type="text"
            placeholder="自定义输入（可选）..."
            value={textInput}
            onChange={(e) => setTextInput(e.target.value)}
            disabled={resolved}
            className="w-full rounded-md border border-[var(--aria-line)] px-3 py-1.5 text-sm text-[var(--aria-ink)] placeholder:text-[var(--aria-ink-muted)] disabled:opacity-60"
          />
        </div>

        {!resolved && onChoiceResponse && (
          <div className="flex justify-end pl-6">
            <button
              type="button"
              onClick={handleSubmit}
              disabled={selected.size === 0 && !textInput.trim()}
              className="inline-flex h-8 items-center rounded-md bg-[var(--aria-primary)] px-4 text-xs font-semibold text-white hover:opacity-90 disabled:opacity-50"
            >
              确认选择
            </button>
          </div>
        )}

        {resolved && (
          <div className="pl-6 text-xs text-[var(--aria-ink-muted)]">已选择</div>
        )}
      </div>
    </ChatEntryContainer>
  );
}
```

resolvePermissionRequest: (id: string) => {
